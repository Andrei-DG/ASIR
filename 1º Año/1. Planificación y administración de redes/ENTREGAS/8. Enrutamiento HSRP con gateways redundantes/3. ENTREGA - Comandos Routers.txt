(R0)
enable
configure terminal
interface G0/0
ip add 192.168.1.1 255.255.255.0
standby version 2
standby 1 ip 192.168.1.3
standby 1 priority 150
standby 1 preempt
no shutdown
exit
interface G0/1
ip add 172.16.0.1 255.255.0.0
standby version 2
standby 2 ip 172.16.0.3
standby 2 priority 150
standby 2 preempt
no shutdown
exit
exit
copy running-config startup-config
